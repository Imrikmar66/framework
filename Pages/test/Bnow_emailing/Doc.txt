Note sur l'utilisation du pluggin Bnow_emailing
/* ------ !!! Champs obligatoires !!! ----- */

erreur : balise avec la classe error_info pour la pop up d'erreur
submit : input type= submit pour envoyer + classe bnow_email_send

/*-------- Shortcode -----------*/
[emailing_form to="" template=""] //to : email destinataire, template : nom du template ('base', 'email', ...) sans l'extension (.tpl)
/* Html interne du formulaire. Les balises <form> sont ajoutées automatiquement */
<input type="text" name="firstname" class="isSimpleString" data-length=5 />
[/emailing_form]

/* ----- Paramètre des inputs -------- */
//Le champ est obligatoire
.isRequired
//Vérification des champs par les classes : 
.isLetterString : Chaine de caractère uniquement constitué de lettre (pas d'espace)
.isSimpleString : Chaine de caractère simple
.isMail : EMail
.isPhoneNumber : Numéro de téléphone
.isNumeric : Ne contient que des chiffres

/*------Vérification taille----*/
//On utilise l'attribut data-length
data-length = 10 //10 caractères minimums

//Styles des erreurs
.bnow_emailing_error //classe qui s'ajoute quand erreur sur le champ
.bnow_email_noDisp //classe qui s'ajoute pour display none

/*------- Variable de template ---------*/
$isPJ -> n'existe que si une piece jointe existe -> utiliser isset(isPJ) 

/*----- Exemple creation de formulaire de contact ---------*/
$(document).ready(function(){
    
    $('#form1').bnow_emailing({
            required : 'Veuillez remplir les champs obligatoires',
            dataLength : 'caractères minimum',
            letterString : 'Ce champ ne peut comporter que des lettres',
            simpleString : 'Ce champ ne peut comporter que des caractère alphanumériques',
            mail : 'Le format mail est incorrect',
            phoneNumber : 'Le format numéro de téléphone est incorrect',
            numeric : 'Ce champ ne peut contenir que des caractère numériques',
            onlyLetter : 'Ce champ ne peut comporter que des lettres'
        });
        
});
